    guard !input.isEmpty else { return true }

    let right = input.iterator(input.endIndex)
    --right
    
    while left < right {
        while left < right && !left.character.isLetterOrNumber {
            ++left
        }
        
        while left < right && !right.character.isLetterOrNumber {
            --right
        }
        
        if left >= right { break }

            return false
        }
        
        ++left
        --right
    }
    
    return true
}
